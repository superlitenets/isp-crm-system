declare(ip.path + "X_HW_ServiceList", null, {value: "INTERNET"});

// Configure WiFi 4
declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.Enable", null, {value: true});
declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.X_HW_VLANID", null, {value: 660});
declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.X_HW_APIsolation", null, {value: true});
