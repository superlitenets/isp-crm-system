✅ Production-Safe GenieACS Provision

Create new provision:

Name:

guest_vlan_660_bridge


Paste this:

// ==============================
// CHECK IF VLAN 660 ALREADY EXISTS
// ==============================

var existing = false;

var wans = declare("InternetGatewayDevice.WANDevice.1.WANConnectionDevice.*.WANIPConnection.*.X_HW_VLAN", {value: 1});

for (var i in wans) {
  if (wans[i].value[0] == 660) {
    existing = true;
    break;
  }
}

// ==============================
// CREATE BRIDGE ONLY IF MISSING
// ==============================

if (!existing) {

  // Create WANConnectionDevice dynamically
  var wan = declare("InternetGatewayDevice.WANDevice.1.WANConnectionDevice.", null, {path: 1});

  // Create WANIPConnection dynamically
  var ip = declare(wan.path + "WANIPConnection.", null, {path: 1});

  // Configure bridge
  declare(ip.path + "Enable", null, {value: true});
  declare(ip.path + "ConnectionType", null, {value: "IP_Bridged"});
  declare(ip.path + "NATEnabled", null, {value: false});
  declare(ip.path + "X_HW_VLAN", null, {value: 660});
  declare(ip.path + "X_HW_ServiceList", null, {value: "INTERNET"});
  declare(ip.path + "Name", null, {value: "WIFI_GUEST_VLAN_660"});
}

// ==============================
// CONFIGURE WLAN 4 (SAFE TO REAPPLY)
// ==============================

declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.Enable", null, {value: true});
declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.X_HW_VLANID", null, {value: 660});
declare("InternetGatewayDevice.LANDevice.1.WLANConfiguration.4.X_HW_APIsolation", null, {value: true});
